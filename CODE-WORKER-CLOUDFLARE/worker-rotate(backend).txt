// ENCRYPTION SERVICE - AES-256-GCM
// ============================================================================

class EncryptionService {
    /**
     * Encrypt data using AES-256-GCM
     * @param {ArrayBuffer} data - Data to encrypt
     * @param {CryptoKey} key - Encryption key
     * @returns {Promise<{encrypted: ArrayBuffer, iv: Uint8Array, authTag: Uint8Array}>}
     */
    static async encrypt(data, key) {
        const iv = crypto.getRandomValues(new Uint8Array(16)); // 128-bit IV
        
        const encrypted = await crypto.subtle.encrypt(
            {
                name: 'AES-GCM',
                iv: iv,
                tagLength: 128 // 128-bit auth tag
            },
            key,
            data
        );
        
        // AES-GCM produces encrypted data with auth tag appended
        // Split into encrypted data and auth tag
        const encryptedArray = new Uint8Array(encrypted);
        const authTag = encryptedArray.slice(-16); // Last 16 bytes
        const encryptedData = encryptedArray.slice(0, -16); // All except last 16 bytes
        
        return {
            encrypted: encryptedData.buffer,
            iv: iv,
            authTag: authTag
        };
    }
    
    /**
     * Decrypt data using AES-256-GCM
     * @param {ArrayBuffer} encryptedData - Encrypted data
     * @param {Uint8Array} iv - Initialization vector
     * @param {Uint8Array} authTag - Authentication tag
     * @param {CryptoKey} key - Decryption key
     * @returns {Promise<ArrayBuffer>}
     */
    static async decrypt(encryptedData, iv, authTag, key) {
        // Combine encrypted data and auth tag
        const combined = new Uint8Array(encryptedData.byteLength + authTag.byteLength);
        combined.set(new Uint8Array(encryptedData), 0);
        combined.set(authTag, encryptedData.byteLength);
        
        const decrypted = await crypto.subtle.decrypt(
            {
                name: 'AES-GCM',
                iv: iv,
                tagLength: 128
            },
            key,
            combined
        );
        
        return decrypted;
    }
    
    /**
     * Generate random encryption key (256-bit)
     * @returns {Promise<CryptoKey>}
     */
    static async generateKey() {
        return crypto.subtle.generateKey(
            {
                name: 'AES-GCM',
                length: 256
            },
            true, // extractable
            ['encrypt', 'decrypt']
        );
    }
    
    /**
     * Export key to base64 string
     * @param {CryptoKey} key
     * @returns {Promise<string>}
     */
    static async exportKey(key) {
        const exported = await crypto.subtle.exportKey('raw', key);
        return btoa(String.fromCharCode(...new Uint8Array(exported)));
    }
    
    /**
     * Import key from base64 string
     * @param {string} keyBase64
     * @returns {Promise<CryptoKey>}
     */
    static async importKey(keyBase64) {
        const keyData = Uint8Array.from(atob(keyBase64), c => c.charCodeAt(0));
        return crypto.subtle.importKey(
            'raw',
            keyData,
            {
                name: 'AES-GCM',
                length: 256
            },
            true,
            ['encrypt', 'decrypt']
        );
    }
}

class InstallationManager {
    /**
     * Initialize new installation
     * Creates encrypted image, stores in R2, returns installation metadata
     */
    static async initializeInstallation(kv, r2, env, {
        vpsIp,
        osVersion,
        rdpPort = 3389,
        metadata = {},
        secret,
        secureMode = false // New: enable secure IMG mode with per-user keys
    }) {
        // Generate unique installation ID
        const installId = crypto.randomUUID();
        
        // Get base image URL (for reference, not for encryption)
        const baseImageUrl = BASE_IMAGE_URL_MAP[osVersion];
        if (!baseImageUrl) {
            throw new Error(`Invalid OS version: ${osVersion}`);
        }
        
        // SECURE MODE: Generate per-user keys for encrypted IMG
        let userKey = null;
        let encryptedMasterKey = null;
        
        if (secureMode) {
            // Master key is stored in env.MASTER_KEY (set by admin)
            const masterKey = env.MASTER_KEY;
            if (!masterKey) {
                throw new Error('MASTER_KEY not configured. Please set MASTER_KEY in environment variables.');
            }
            
            // Generate unique user key (random, per installation)
            userKey = crypto.randomBytes(32).toString('base64'); // 256-bit key
            
            // Encrypt master key with user key (AES-256-CBC with PBKDF2)
            const salt = crypto.randomBytes(16);
            const iv = crypto.randomBytes(16);
            const key = crypto.pbkdf2Sync(userKey, salt, 100000, 32, 'sha256');
            const cipher = crypto.createCipheriv('aes-256-cbc', key, iv);
            const encrypted = Buffer.concat([
                cipher.update(masterKey, 'utf8'),
                cipher.final()
            ]);
            
            // Format: salt (16) + iv (16) + encrypted data
            encryptedMasterKey = Buffer.concat([salt, iv, encrypted]).toString('base64');
            
            console.log(`[INSTALL] Secure mode: Generated user key and encrypted master key`);
            console.log(`[INSTALL] User key: ${userKey.substring(0, 16)}...`);
        }
        
        // Use rdp-installer-azovest.img from public URL 
        // Binary file is available at: https://api.eov.my.id/azovest/rdp-installer-azovest.img
        const binaryInstallerUrl = env.BINARY_INSTALLER_URL || 'https://api.eov.my.id/azovest/rdp-installer-azovest.img';
        
        // Generate token for binary installer URL (like other URLs)
        const binaryInstallerToken = await generateToken(kv, 'encrypted-binary', {
            url: binaryInstallerUrl,
            installId,
            vpsIp
        });
        
        console.log(`[INSTALL] Using binary installer from: ${binaryInstallerUrl}`);
        console.log(`[INSTALL] Generated token for binary installer: ${binaryInstallerToken.substring(0, 16)}...`);
        
        // Generate token for OS image (tele.sh needs this)
        // MUST bind to VPS IP target (required for security)
        const imageToken = await generateToken(kv, 'image', {
            url: baseImageUrl,
            osVersion,
            vpsIp // Bind to VPS IP target
        });
        
        console.log(`[INSTALL] Generated token for OS image: ${imageToken.substring(0, 16)}...`);
        
        // Store installation metadata in KV
        const installationData = {
            installId,
            vpsIp,
            osVersion,
            rdpPort,
            baseImageUrl, // OS image URL (for reference)
            binaryInstallerUrl, // Original binary installer URL (for backend to download)
            binaryInstallerToken, // Token for binary installer URL
            imageToken, // Token for OS image (for tele.sh)
            secureMode, // Flag for secure IMG mode
            userKey, // User key (if secure mode)
            encryptedMasterKey, // Encrypted master key (if secure mode)
            status: 'ready',
            used: false,
            keyUsed: false, // Track if encryption key has been used for this installation (one-time use)
            downloadCount: 0,
            maxDownloads: 1, // One-time use only
            createdAt: Date.now(),
            expiresAt: Date.now() + (30 * 60 * 1000), // 30 minutes
            metadata
        };
        
        // Store in KV with TTL
        await kv.put(`installation:${installId}`, JSON.stringify(installationData), {
            expirationTtl: 1800 // 30 minutes
        });
        
        console.log(`[INSTALL] Initialized installation ${installId} for ${vpsIp}, OS: ${osVersion}`);
        
        const response = {
            installId,
            downloadUrl: `/i-enc/${binaryInstallerToken}`, // VPS will download .img from this tokenized URL
            imageToken, // Token for OS image (for tele.sh)
            status: 'ready',
            expiresAt: installationData.expiresAt,
            message: 'Installation initialized. VPS should download .img from downloadUrl (tokenized).'
        };
        
        // Add secure mode keys if enabled
        if (secureMode && userKey && encryptedMasterKey) {
            response.userKey = userKey;
            response.encryptedMasterKey = encryptedMasterKey;
            response.secureMode = true;
        }
        
        return response;
    }
    
    /**
     * Get binary installer URL for bot server to download and encrypt
     * Bot server will download rdp-installer-azovest.img, encrypt it, and upload to R2
     */
    static async getBinaryInstallerUrl(installationData) {
        const { binaryInstallerUrl } = installationData;
        return {
            binaryInstallerUrl,
            message: 'Bot server should download rdp-installer-azovest.img, encrypt it, and upload to R2'
        };
    }
    
    /**
     * Check if encrypted image exists in R2
     * Bot server uploads encrypted image after encryption
     */
    static async checkEncryptedImage(r2, installId) {
        if (!r2) {
            throw new Error('R2 bucket not configured. Please bind ENCRYPTED_IMAGES R2 bucket to worker.');
        }
        
        const r2Key = `encrypted/${installId}.img.enc`;
        try {
            const existing = await r2.head(r2Key);
            if (existing) {
                console.log(`[INSTALL] Encrypted image exists in R2: ${r2Key}`);
                return { exists: true, r2Key };
            }
        } catch (headError) {
            // File doesn't exist
        }
        
        return { exists: false, r2Key };
    }
    
    /**
     * Get installation data
     */
    static async getInstallation(kv, installId) {
        const data = await kv.get(`installation:${installId}`);
        if (!data) return null;
        return JSON.parse(data);
    }
    
    /**
     * Update installation status
     */
    static async updateInstallation(kv, installId, updates) {
        const installation = await this.getInstallation(kv, installId);
        if (!installation) {
            throw new Error('Installation not found');
        }
        
        const updated = { ...installation, ...updates, updatedAt: Date.now() };
        
        const ttl = Math.floor((updated.expiresAt - Date.now()) / 1000);
        if (ttl > 0) {
            await kv.put(`installation:${installId}`, JSON.stringify(updated), {
                expirationTtl: ttl
            });
        }
        
        return updated;
    }
    
    /**
     * Cleanup installation (delete from R2 and KV)
     */
    static async cleanup(kv, r2, installId) {
        console.log(`[INSTALL] Cleaning up installation ${installId}...`);
        
        // Get installation data
        const installation = await this.getInstallation(kv, installId);
        
        // Delete from R2
        if (r2) {
            const r2Key = `encrypted/${installId}.img.enc`;
            try {
                await r2.delete(r2Key);
                console.log(`[INSTALL] Deleted encrypted image from R2: ${r2Key}`);
            } catch (e) {
                console.error(`[INSTALL] Failed to delete from R2: ${e.message}`);
            }
        }
        
        // Delete encryption key and mark as cleaned up
        if (installation) {
            const cleaned = {
                ...installation,
                encryptionKey: null,
                status: 'cleaned',
                cleanedAt: Date.now()
            };
            
            // Store for 24 hours for audit purposes, then auto-delete
            await kv.put(`installation:${installId}`, JSON.stringify(cleaned), {
                expirationTtl: 86400 // 24 hours
            });
        }
        
        console.log(`[INSTALL] Cleanup completed for ${installId}`);
        return { success: true, message: 'Installation cleaned up' };
    }
}

// Token storage menggunakan Cloudflare KV
// Token data: { type, url, expiresAt, used, createdAt }
async function generateToken(kv, type, data) {
    const token = generateRandomToken(); // 64 hex chars (256 bits)
    const expiresAt = Date.now() + (30 * 60 * 1000); // 30 minutes
    
    const tokenData = {
        type,
        data,
        expiresAt,
        used: false,
        useCount: 0, // Track jumlah penggunaan (untuk image token: max 2x)
        maxUses: type === 'image' ? 3 : (type === 'encrypted-binary' ? 1 : (type === 'confhome' ? 30 : 1)), // Image token: 2x use, encrypted-binary: 1x (one-time use), confhome: 100x (many variants), lainnya: 1x
        createdAt: Date.now()
    };
    
    // Store in KV with TTL (expires in 30 minutes)
    await kv.put(`token:${token}`, JSON.stringify(tokenData), {
        expirationTtl: 1800 // 30 minutes in seconds
    });
    
    return token;
}

async function validateToken(kv, token, type, clientIp = null) {
    const tokenKey = `token:${token}`;
    console.info(`[VALIDATE] Looking for key: ${tokenKey}, type: ${type}`);
    const tokenDataStr = await kv.get(tokenKey);
    console.info(`[VALIDATE] KV get result: ${tokenDataStr ? 'FOUND' : 'NOT FOUND'}`);
    
    if (!tokenDataStr) {
        return { valid: false, error: 'Token tidak ditemukan' };
    }
    
    const tokenData = JSON.parse(tokenDataStr);
    
    if (Date.now() > tokenData.expiresAt) {
        await kv.delete(tokenKey);
        return { valid: false, error: 'Token expired' };
    }
    
    if (tokenData.type !== type) {
        return { valid: false, error: 'Token type mismatch' };
    }
    
    // IP binding validation: semua token yang punya vpsIp HARUS match dengan clientIp
    if (tokenData.data && tokenData.data.vpsIp && clientIp) {
        if (clientIp !== tokenData.data.vpsIp) {
            console.error(`[VALIDATE] IP mismatch: clientIp=${clientIp}, expected vpsIp=${tokenData.data.vpsIp}`);
            return { valid: false, error: `IP address ${clientIp} is not authorized. Only VPS target IP ${tokenData.data.vpsIp} can access.` };
        }
    }
    
    // Image token: boleh digunakan 2x (untuk HEAD request + GET request dari dd raw)
    // Encrypted-binary token: one-time use (untuk security)
    // Confhome token: 100x use (banyak variant file: debian.cfg, ubuntu.yaml, redhat.cfg, dll)
    // Script token: one-time use (untuk security)
    const maxUses = tokenData.maxUses || (type === 'image' ? 2 : (type === 'encrypted-binary' ? 1 : (type === 'confhome' ? 100 : 1)));
    const useCount = tokenData.useCount || 0;
    
    if (useCount >= maxUses) {
        return { valid: false, error: `Token sudah digunakan ${useCount} kali (max ${maxUses}x use)` };
    }
    
    // Increment use count
    tokenData.useCount = useCount + 1;
    tokenData.used = tokenData.useCount >= maxUses; // Mark as used jika sudah mencapai max uses
    
    // Update token di KV
    await kv.put(tokenKey, JSON.stringify(tokenData), {
        expirationTtl: Math.floor((tokenData.expiresAt - Date.now()) / 1000)
    });
    
    // Delete setelah expired atau jika sudah max uses
    if (type === 'image') {
        // Image token: delete setelah 5 menit atau setelah expired (allow time untuk download)
        // Tapi jangan delete terlalu cepat jika masih dalam download process
        if (tokenData.used) {
            // Sudah 2x use, delete setelah 5 menit (cukup untuk download besar)
            setTimeout(() => {
                kv.delete(tokenKey).catch(() => {});
            }, 5 * 60 * 1000); // 5 menit grace period untuk download besar
        }
    } else if (type === 'confhome') {
        // Confhome token: delete setelah expired atau setelah 100x use (banyak variant file)
        // Jangan delete terlalu cepat karena banyak file yang perlu diakses
        if (tokenData.used) {
            // Sudah 100x use, delete setelah 10 menit (cukup untuk akses banyak file)
            setTimeout(() => {
                kv.delete(tokenKey).catch(() => {});
            }, 10 * 60 * 1000); // 10 menit grace period untuk akses banyak file
        }
    } else {
        // Script/encrypted-binary token: delete setelah use (one-time use)
        if (tokenData.used) {
            kv.delete(tokenKey).catch(() => {});
        }
    }
    
    return { valid: true, data: tokenData.data };
}

function generateRandomToken() {
    // Generate 32 random bytes (256 bits) and convert to hex (64 chars)
    const array = new Uint8Array(32);
    crypto.getRandomValues(array);
    return Array.from(array, byte => byte.toString(16).padStart(2, '0')).join('');
}

        unusedTokens: 0
    };
}

                    status: testResponse.status,
                    statusText: testResponse.statusText,
                    fetchTimeMs: fetchTime,
                    method: method,
                    url: imageUrl,
                    headers: headers,
                    bodyPreview: bodyPreview.substring(0, 200),
                    analysis: {
                        isHtml: bodyPreview.toLowerCase().includes('<html'),
                        isJson: bodyPreview.trim().startsWith('{'),
                        isBinary: !bodyPreview.toLowerCase().includes('<html') && !bodyPreview.trim().startsWith('{'),
                        message: isError ? '❌ Response is error' : '✅ Response OK'
                    }
                }, null, 2), {
                    status: 200,
                    headers: { 
                        'Content-Type': 'application/json',
                        'Access-Control-Allow-Origin': '*'
                    }
                });
                
            } catch (error) {
                return new Response(JSON.stringify({
                    success: false,
                    error: error.message,
                    stack: error.stack,
                    url: imageUrl
                }, null, 2), {
                    status: 200,
                    headers: { 
                        'Content-Type': 'application/json',
                        'Access-Control-Allow-Origin': '*'
                    }
                });
            }
        }
        
        // Get KV namespace and R2 bucket (must be bound in wrangler.toml)
        const kv = env.TOKEN_STORAGE;
        const r2 = env.ENCRYPTED_IMAGES; // R2 bucket for encrypted images
        
        // CORS headers + Security headers untuk prevent URL exposure
        const corsHeaders = {
            'Access-Control-Allow-Origin': '*',
            'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
            'Access-Control-Allow-Headers': 'Content-Type, X-Api-Secret',
            // Security headers untuk prevent browser prefetch/preconnect ke URL asli
            'X-DNS-Prefetch-Control': 'off', // Disable DNS prefetch
            'X-Content-Type-Options': 'nosniff', // Prevent MIME type sniffing
        };
        
        // Handle OPTIONS (CORS preflight)
        if (method === 'OPTIONS') {
            return new Response(null, {
                headers: corsHeaders
            });
        }
        
        try {
            // Health check and version info
            if (path === '/health' && method === 'GET') {
                const stats = await getTokenStats(kv);
                return new Response(JSON.stringify({ 
                    status: 'ok', 
                    version: '2.0.0-fixed', // Version to identify this is the correct worker
                    timestamp: Date.now(),

                console.log(`[CONFHOME] Access request with token: ${token.substring(0, 8)}...`);
                console.log(`  Client IP: ${clientIp}`);
                console.log(`  VPS IP (target): ${vpsIp}`);
                
                if (vpsIp && clientIp !== vpsIp) {
                    console.error(`[CONFHOME] Access denied: IP=${clientIp} (expected VPS IP: ${vpsIp})`);
                    return new Response(JSON.stringify({ 
                        error: 'Forbidden',
                        message: `IP address ${clientIp} is not authorized. Only VPS target IP ${vpsIp} can access.`
                    }), {
                        status: 403,
                        headers: {
                            'Content-Type': 'application/json',
                            ...corsHeaders
                        }
                    });
                }
                
                // Extract filename from path (e.g., /c/token/fix-eth-name.sh -> fix-eth-name.sh)
                // Token format: /c/TOKEN or /c/TOKEN/filename
                let pathAfterToken = '';
                if (path.startsWith('/c/')) {
                    const parts = path.split('/c/')[1].split('/');
                    if (parts.length > 1) {
                        // Skip token (first part), take rest as file path
                        pathAfterToken = parts.slice(1).join('/');
                    }
                } else if (path.startsWith('/api/confhome/')) {
                    const parts = path.split('/api/confhome/')[1].split('/');
                    if (parts.length > 1) {
                        // Skip token (first part), take rest as file path
                        pathAfterToken = parts.slice(1).join('/');
                    }
                }
                
                // Build URL: confhome base URL + filename
                const confhomeBaseUrl = validation.data.url;
                // Ensure confhomeBaseUrl ends without slash, then add pathAfterToken
                const baseUrlClean = confhomeBaseUrl.replace(/\/+$/, '');
                const fileUrl = pathAfterToken 
                    ? `${baseUrlClean}/${pathAfterToken}`.replace(/\/+/g, '/')
                    : baseUrlClean;
                
                console.log(`[CONFHOME] Proxying request to: ${fileUrl.replace(/https?:\/\/([^\/]+).*/, '$1/***')}`);
                
                // PROXY MODE: Fetch dan stream file content langsung (tidak redirect)
                try {
                    const fileResponse = await fetch(fileUrl, {
                        method: 'GET',
                        redirect: 'follow',
                        headers: {
                            'User-Agent': 'Mozilla/5.0 (compatible; RDP-Installer/1.0)',
                            'Accept': '*/*',
                            'Referer': ''
                        }
                    });
                    
                    if (!fileResponse.ok) {
                        console.error(`[CONFHOME] Failed to fetch file: ${fileResponse.status} ${fileResponse.statusText}`);
                        return new Response(`Failed to fetch file: ${fileResponse.status}`, {
                            status: fileResponse.status,
                            headers: {
                                'Content-Type': 'text/plain; charset=utf-8',
                                ...corsHeaders
                            }
                        });
                    }
                    
                    // Stream file content langsung ke client
                    const contentType = fileResponse.headers.get('Content-Type') || 'application/octet-stream';
                    
                    return new Response(fileResponse.body, {
                        status: 200,
                        headers: {
                            'Content-Type': contentType,
                            'Cache-Control': 'no-store, no-cache, must-revalidate, private',
                            'Pragma': 'no-cache',
                            'Expires': '0',
                            ...corsHeaders
                        }
                    });
                } catch (fetchError) {
                    console.error(`[CONFHOME] Error fetching file: ${fetchError.message}`);
                    return new Response(`Error fetching file: ${fetchError.message}`, {
                        status: 500,
                        headers: {
                            'Content-Type': 'text/plain; charset=utf-8',
                            ...corsHeaders
                        }
                    });
                }
            }
            
            // Installation status API: POST /api/installation-status/:installId
            // Untuk tracking status instalasi dari trans.sh sebelum reboot
            if (path.startsWith('/api/installation-status/') && method === 'POST') {
                const installId = path.split('/api/installation-status/')[1];
                if (!installId) {
                    return new Response(JSON.stringify({ error: 'Install ID required' }), {
                        status: 400,
                        headers: {
                            'Content-Type': 'application/json',
                            ...corsHeaders
                        }
                    });
                }
                
                // Parse request body
                let statusData;
                try {
                    statusData = await request.json();
                } catch (e) {
                    statusData = {};
                }
                
                const status = statusData.status || 'completed'; // 'completed', 'failed', 'in_progress'
                const message = statusData.message || '';
                
                // Check if KV is available
                if (!kv) {
                    console.error('[INSTALLATION STATUS] KV namespace TOKEN_STORAGE not bound');
                    return new Response(JSON.stringify({ 
                        error: 'Backend configuration error: KV namespace not bound' 
                    }), {
                        status: 500,
                        headers: {
                            'Content-Type': 'application/json',
                            ...corsHeaders
                        }
                    });
                }
                
                // Store installation status in KV
                // Key: installation:${installId}
                // Data: { status, message, timestamp, updatedAt }
                const statusKey = `installation:${installId}`;
                const statusInfo = {
                    installId,
                    status,
                    message,
                    timestamp: Date.now(),
                    updatedAt: Date.now()
                };
                
                // Store with 1 hour TTL (installation should complete within 1 hour)
                await kv.put(statusKey, JSON.stringify(statusInfo), {
                    expirationTtl: 3600 // 1 hour
                });
                
                console.info(`[INSTALLATION STATUS] Updated status for ${installId}: ${status}${message ? ` - ${message}` : ''}`);
                
                return new Response(JSON.stringify({
                    success: true,
                    installId,
                    status,
                    message: 'Status updated successfully'
                }), {
                    headers: {
                        'Content-Type': 'application/json',
                        ...corsHeaders
                    }
                });
            }
            
            // Get installation status: GET /api/installation-status/:installId
            // Untuk polling status dari dedicatedRdpHandler.js
            if (path.startsWith('/api/installation-status/') && method === 'GET') {
                const installId = path.split('/api/installation-status/')[1];
                if (!installId) {
                    return new Response(JSON.stringify({ error: 'Install ID required' }), {
                        status: 400,
                        headers: {
                            'Content-Type': 'application/json',
                            ...corsHeaders
                        }
                    });
                }
                
                // Check if KV is available
                if (!kv) {
                    console.error('[INSTALLATION STATUS] KV namespace TOKEN_STORAGE not bound');
                    return new Response(JSON.stringify({ 
                        error: 'Backend configuration error: KV namespace not bound' 
                    }), {
                        status: 500,
                        headers: {
                            'Content-Type': 'application/json',
                            ...corsHeaders
                        }
                    });
                }
                
                // Get installation status from KV
                const statusKey = `installation:${installId}`;
                const statusDataStr = await kv.get(statusKey);
                
                if (!statusDataStr) {
                    // Status not found (belum ada status atau sudah expired)
                    return new Response(JSON.stringify({
                        success: true,
                        installId,
                        status: 'not_found',
                        message: 'Installation status not found or expired'
                    }), {
                        headers: {
                            'Content-Type': 'application/json',
                            ...corsHeaders
                        }
                    });
                }
                
                const statusData = JSON.parse(statusDataStr);
                
                return new Response(JSON.stringify({
                    success: true,
                    installId,
                    status: statusData.status,
                    message: statusData.message || '',
                    timestamp: statusData.timestamp,
                    updatedAt: statusData.updatedAt
                }), {
                    headers: {
                        'Content-Type': 'application/json',
                        ...corsHeaders
                    }
                });
            }
            
            // 404
            return new Response(JSON.stringify({ error: 'Not found' }), {
                status: 404,
                headers: {
                    'Content-Type': 'application/json',
                    ...corsHeaders
                }
            });
            
        } catch (error) {
            console.error('Error:', error);
            return new Response(JSON.stringify({ 
                error: 'Internal server error',
                message: error.message 
            }), {
                status: 500,
                headers: {
                    'Content-Type': 'application/json',
                    ...corsHeaders
                }
            });
        }
    }
};
